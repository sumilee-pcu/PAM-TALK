async function redeemCollateral(userAddress, dcToBurn) {

    from: userAddress,

  collateral.algoAmount -= algoToReturn;

  if (collateral.dcMinted === 0) {

  // 7. ë°ì´í„°ë² ì´ìŠ¤ ê¸°ë¡

    return_tx_id: returnTxId,
    redeemed_at: Date.now()
  });

  return {
    success: true,
    dcBurned: dcToBurn,
    algoReturned: algoToReturn,
    burnTxId,
    returnTxId
  };
